411  static void compileTableEntry(size_t row, size_t diag, vector<vector<tokens>>& table) {
412  for (size_t i(0); i < diag; i++) {
413  tokens first(getUClosure(table[row][i]));
414  for (unsigned char fi(0); fi < first.size(); fi++) {

434  parser(u32string const& re, literals const& lits) {
435  if (re.empty()) {
436  badRegularExpression = true;
437  return;
438  }
439  symbolMappingIndex = 0;
440  size_t numberOfTokens(re.length());
441  symbolMapping.reserve(numberOfTokens);
442  table.reserve(numberOfTokens);
443  size_t row(0);
444  for (char32_t symbol : re) {
445  table.push_back(vector<tokens>(numberOfTokens-row, tokens()));
446  if (symbol == lits.L) { table[row][0].set(token::L); }
447  else if (symbol == lits.R) { table[row][0].set(token::R); }
448  else if (symbol == lits.P) { table[row][0].set(token::P); }
449  else if (symbol == lits.S) { table[row][0].set(token::S); }
450  else {
451  table[row][0].set(token::Σ);
452  symbolMapping.push_back(symbol);
453  }
454  row++;
455  }
456  symbolMapping.shrink_to_fit();
457  for (size_t diag(1); diag < numberOfTokens; diag++) {
458  for (size_t row(0); row < numberOfTokens - diag; row++) {
459  compileTableEntry(row, diag, table);
460  }
461  }
462  if (!getUClosure(table[0][table[0].size()-1])[token::A]) {

501  tree(size_t row, size_t diag, parser const* p) :
502  p(p),
503  symbol([&]()->token{
504  for (unsigned char i(token::END); i > 0; i--) {
505  if (p->table[row][diag][i-1]) {
506  return static_cast<token>(i-1);

514  exptr operator()(bool optimized, bool aggressive) {
515  switch (symbol) {
516  case token::A:
517  return spawnAlternation((*children.first)(optimized, aggressive), (*children.second)(optimized, aggressive), optimized, aggressive);
518  case token::B:
519  return (*children.second)(optimized, aggressive);
520  case token::C:
521  return spawnConcatenation((*children.first)(optimized, aggressive), (*children.second)(optimized, aggressive), optimized, aggressive);
522  case token::K:
523  return spawnKleene((*children.first)(optimized, aggressive), optimized, aggressive);
524  case token::E:
525  return (*children.second)(optimized, aggressive);
526  case token::F:
527  return (*children.first)(optimized, aggressive);
528  case token::Σ: {
529  char32_t symbol = p->symbolMapping[p->symbolMappingIndex++];
530  if (p->symbolMappingIndex >= p->symbolMapping.size()) { p->symbolMappingIndex = 0; }
531  if (symbol == p->lits.EPSILON) { return spawnEmptyString(); }
532  else if (symbol == p->lits.EMPTY) { return spawnEmptySet(); }
533  else { return spawnSymbol(symbol); }
534  }
535  default:
536  return exptr();
537  }
538  }
