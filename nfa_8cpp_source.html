478 nfa nfa::builder::build() {
479  p->alphabet.insert(U'\0');
480  vector<char32_t> alphabetV(p->alphabet.begin(), p->alphabet.end());
481  std::sort(alphabetV.begin(), alphabetV.end());
482  vector<string> labelV(p->transitions.size());
483  labelV[0] = p->initial;
484  valarray<bool> acceptingV(p->transitions.size());
485  acceptingV[0] = (p->acceptingStates.find(p->initial) != p->acceptingStates.end());
486  size_t i(1);
487  for (auto entry : p->transitions) {
488  if (entry.first == p->initial) {
489  continue;
490  }
491  labelV[i] = entry.first;
492  acceptingV[i++] = (
493  p->acceptingStates.find(entry.first) != p->acceptingStates.end()
494  );
495  }
496  vector<vector<valarray<bool>>> transitionV(
497  p->transitions.size(),
498  vector<valarray<bool>>(
499  p->alphabet.size()+1,
500  valarray<bool>(labelV.size())
501  )
502  );
503  unordered_set<string> emptySet;
504  size_t qi(0);
505  for (auto const& q : labelV) {
506  unordered_map<char32_t,unordered_set<string>> const& fromQ = p->transitions[q];
507  size_t si(0);
508  for (char32_t symbol : alphabetV) {
509  auto to = fromQ.find(symbol);
510  unordered_set<string> const& viaSymbol(to != fromQ.end() ? to->second : emptySet);
511  i = 0;
512  for (auto const& p : labelV) {
513  transitionV[qi][si][i++] = (viaSymbol.count(p) != 0);
514  }
515  si++;
516  }
517  qi++;
518  }
519  return {alphabetV, transitionV, labelV, acceptingV};
520 }
